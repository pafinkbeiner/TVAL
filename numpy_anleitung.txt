import numpy as np
from PIL import Image

# Öffne das Bild als NumPy-Array
im = np.array(Image.open('my_image.jpg'))

# Extrakte einen Bereich aus dem Bild mit einer Größe von 100x100 Pixeln,
# der sich in der Mitte des Bildes befindet
rows, cols, _ = im.shape
start_row = rows // 2 - 50
end_row = rows // 2 + 50
start_col = cols // 2 - 50
end_col = cols // 2 + 50
cropped_im = im[start_row:end_row, start_col:end_col, :]

# Speichere das zugeschnittene Bild
Image.fromarray(cropped_im).save('cropped_image.jpg')
